<task type="auto">
  <name>Task 2: Create service worker for offline caching</name>
  <files>public/sw.js</files>
  <action>
Create `public/` directory (does not exist yet) and add `public/sw.js`.

Expo copies everything in `public/` to the `dist/` output directory during `expo export`, so `public/sw.js` becomes `dist/sw.js` in the deployed build.

Service worker implementation:

```javascript
const CACHE_NAME = 'lingolock-v1';

// Install: pre-cache app shell
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      // Cache the root HTML — JS/CSS bundles will be cached on first fetch
      return cache.addAll(['/']);
    })
  );
  // Activate immediately without waiting for existing tabs to close
  self.skipWaiting();
});

// Activate: clean up old caches
self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys().then((names) =>
      Promise.all(
        names
          .filter((name) => name !== CACHE_NAME)
          .map((name) => caches.delete(name))
      )
    )
  );
  // Take control of all open tabs immediately
  self.clients.claim();
});

// Fetch: cache-first for static assets, network-first for navigation
self.addEventListener('fetch', (event) => {
  const { request } = event;

  // Skip non-GET requests
  if (request.method !== 'GET') return;

  // Navigation requests: network-first (ensures fresh HTML on deploy)
  if (request.mode === 'navigate') {
    event.respondWith(
      fetch(request)
        .then((response) => {
          const clone = response.clone();
          caches.open(CACHE_NAME).then((cache) => cache.put(request, clone));
          return response;
        })
        .catch(() => caches.match(request))
    );
    return;
  }

  // Static assets (JS, CSS, images, fonts): cache-first
  event.respondWith(
    caches.match(request).then((cached) => {
      if (cached) return cached;
      return fetch(request).then((response) => {
        // Only cache successful responses
        if (response.ok) {
          const clone = response.clone();
          caches.open(CACHE_NAME).then((cache) => cache.put(request, clone));
        }
        return response;
      });
    })
  );
});
```

Key design decisions:
- `skipWaiting()` + `clients.claim()` — new service worker activates immediately on deploy
- Navigation: network-first so users get fresh HTML after Vercel deploys
- Static assets: cache-first for speed (JS/CSS bundles are content-hashed by Metro, so new deploys get new URLs)
- Cache name versioned (`lingolock-v1`) — bump version to force full cache invalidation if needed
- No pre-caching of JS/CSS bundles (their filenames include hashes, can't predict at authoring time) — they get cached on first fetch
